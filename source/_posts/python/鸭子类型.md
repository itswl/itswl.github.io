```
class Duck: 
    @classmethod
    def fly(cls): 
        print("Duck flying") 


class Airplane:
    @staticmethod
    def fly(): 
        print("Airplane flying") 
 
class Bird:
    def fly(self):
        print('Bird is flying')
 

def lift_off(entity): 
    entity.fly() 

if __name__ == "__main__":
    lift_off(Duck)
    lift_off(Airplane)
    bird = Bird()
    lift_off(bird)
    
 ```
